                            GIT

cd <name folder> - переход в папку
cd ../ - переход в папку назад
ls - содержимое папки
clear - очистка консоли
git init - инициализация папки гитом
git add . - добавление всех измененных файлов в ожидание
git commit -m "текст комментария" - комментирование всех файлов находящихся в ожидании
git status - статус измененных файлов
git log --oneline - сокращенный список коммитов в этой папке
git checkout <№ коммита> - атом показывает папку на момент сохранения данного коммита
git reset <# коммита> - удаляет коммит, не меняя файлов
git reset <# коммита> --hard - удаление коммитов и файлов до коммита
